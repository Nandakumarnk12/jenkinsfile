            }
        }
        stage('Build') {
            steps {
               bat 'mvn install'
            }
        }
        stage('Test') {
            steps {
                echo 'This is test stage'
            }
        }
        stage('deploy') {
            steps {
                echo 'This is deploy stage'
            }
        }
    }
}
